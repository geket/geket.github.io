Show Recent Activity

Display your recent GitHub activity feed using the Events API:

// Fetch recent activity

fetch(`https://api.github.com/users/${username}/events/public?per_page=5`)
  .then(res => res.json())
  .then(events => {
    const activity = events.map(e => {
      const date = new Date(e.created_at).toLocaleDateString();
      switch(e.type) {
        case 'PushEvent':
          return `Pushed to ${e.repo.name} on ${date}`;
        case 'CreateEvent':
          return `Created ${e.payload.ref_type} in ${e.repo.name}`;
        case 'PullRequestEvent':
          return `${e.payload.action} PR in ${e.repo.name}`;
        default:
          return `${e.type} in ${e.repo.name}`;
      }
    });
    console.log(activity);
  });
